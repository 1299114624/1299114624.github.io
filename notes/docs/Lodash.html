/** 
 * @desc 数据处理工具
 */
import numeral from 'numeral'
import I18N from '@/lang/i18n'
/** 
 * 数字处理
 */
const NumberHandler = {
    /** 
     * 处理较大数字
     * @param {*} input 输入
     * @param {Boolean} isRangeFormat 是否进行范围处理，默认进行范围处理
     * @return {String} 输出
     */
    handleBigData: function(input, isRangeFormat) {
        const num = parseInt(input)
        if (curLanguage === 'en_US') {
            return numeral(num).format('0,0')
        } else {
            if (num < 10000) {
                return isRangeFormat !== false ? num : numeral(num).format('0.00')+'万'
            }
        }
    },
    /** 
     * 处理数组中较大数字
     * @param {Array} arr 输入
     * @return {String} 输出
     */
    handleBigDataArr: function(arr) {
        const result = []
        for (const item of arr) {
            result.push(this.handleBigData(item))
        }
        return result
    },
    /** 
     * 负数处理
     * @param {String} num 返回数字
     * @param {String} specReturn 负数返回的特殊值
     * @return {String} 处理后文本
     */
    cleanNegaData: function(arnum, specReturn) {
        if (num === -1 || typeof num === 'string' && Number(num) < 0) {
            return specReturn
        }
        return num
    },
    /** 
     * 流量处理
     * @param {String} traffic 流量
     * @return {String} 处理后文本
     */
    transTraffic: function(traffic) {
        if (!traffic || traffic === '' || traffic < 0) {
            return '0M'
        }
        if (traffic < 1000) {
            return _.round(traffic,2) + 'M'
        } else if (traffic < 1000000) {
            return _.round(traffic/1000, 2) + 'G'
        } else {
            return _.round(traffic / 1000000, 2) + 'T'
        }
    },
    /** 
     * 文本格式化，若为空或 -1，返回'-'
     * @param {(String | Number)} num 需要格式化的文本
     * @return {(String | Number)} tempResult 处理后的文本
     */
    formatUndefinedData(num) {
        if (typeof num === 'undefined' || num === -1) {
            return '-'
        } else {
            let tempResult = num
            let tempNum = `${num}`
            tempNum = tempNum.trim()
            const digitReg = /^(-|\+)?\d+(\.\d+)?$/
            if (digitReg.test(tempNum)) {
                tempResult = this.handleBigData(tempNum)
            }
            return tempResult
        }
    },
}

/**
 * 字符串处理
 */
const StringHandler = {
    /** 
     * 处理坐标轴过长字符串
     * @param {String} input 长字符串
     * @return {String} 精简后字符串
     */
    handleChartLongText: function(input) {
        if (!input) {
            return ''
        }
        if (curLanguage === 'zh_CN') {
            if (input.length > 7) {
                return input.substr(0,7) + '...'
            }
        } else if (curLanguage === 'en_US') {
            if (input.length > 10) {
                return input.substr(0,10) + '...'
            }
        }
        return input
    },
    /** 
     * 连接字符串
     * @param {String} str 待连接
     * @param {String} add 添加
     * @param {Array} dir 中英文组装
     * @desc dir[0]:中文 dir[1]:英文 0:after, 1:before
     * @returns {String} 处理的字符串
     */
    concatName: function(str, add, dir=[0,0]) {
        const arr = [str + add, add + str]
        if (curLanguage === 'zh_CN') {
            return arr[dir[0]]
        }
        return arr[dir[1]]
    },
}

/**
 * 路径处理
 */
const UrlHandler = {
    /** 
     * 获取路径后面带过来的参数
     * @param {String} name 匹配参数
     * @return {*} 对应参数的匹配值
     */
    getUrlParam: function(name) {
        // 构造一个含有目标参数的正则表达式对象
        const reg = new RegExp('(^|&)'+name+'=([^&]*)(&|$)')
        const r = window.location.search.substr(1).match(reg) //匹配目标参数、
        const param = r ? decodeURIComponent(r[2]) : null
        return param
    },
}

/**
 * 业务处理
 */
const BusinessHandler = {
    /** 
     * 直接获取I18N对应值
     * @param {String} text 入值
     * @param {String} defaultVal 默认值
     * @return {String} 国际化
     */
    getI18NDirect: function(text, defaultVal) {
        if (!text) {
            return defaultVal || '-'
        }
        const upperCaseText = text.toUpperCase()
        return I18N[upperCaseText] || defaultVal || '-'
    },
}

export {
    NumberHandler, // 数字处理
    StringHandler, // 字符串处理
    UrlHandler, // 路径处理
    BusinessHandler // 业务相关特殊处理
}
